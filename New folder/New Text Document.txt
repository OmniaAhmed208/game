public function handleGoogleCallback()
{
    $user = Socialite::driver('google')->user();

    // Get the user's access token
    $accessToken = $user->token;

    // Create a Google client and set the access token
    $client = new Google_Client();
    $client->setAccessToken($accessToken);

    // Create a YouTube service using the client
    $youtube = new Google_Service_YouTube($client);

    // Replace 'CHANNEL_ID' with the channel ID you want to access
    $channelId = 'CHANNEL_ID';

    // Make a request to the YouTube Data API to fetch the list of videos from the channel
    $videos = $youtube->search->listSearch('snippet', [
        'channelId' => $channelId,
        'maxResults' => 10, // Change this number as needed
    ]);

    // You can now use the $videos object to access the video data
    $videoList = $videos->getItems();

    // Do something with the video data
    foreach ($videoList as $video) {
        echo $video->snippet->title . '<br>';
    }
}
